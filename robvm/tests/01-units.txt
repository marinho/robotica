UNITS
=====

    >>> from robvm.units import ohm, volt, ampere, coulomb, watt, unknown, second, Value
    >>> from decimal import Decimal

    >>> issubclass(Value, Decimal)
    True

    >>> Value('1 coulomb')
    1 coulomb
    
Converts a coulumb to eletrons

    >>> Value('3 coulomb').as_eletrons()
    18.726x10^18

I = Q / t

    >>> (Value('3 coulomb') / Value('2 second')).as_string()
    '1.5 ampere'

t = Q / I

    >>> (Value('3 coulomb') / Value('1.5 ampere')).as_string()
    '2 second'

Q = t * I

    >>> (Value('5 second') * Value('0.5 ampere')).as_string()
    '2.5 coulomb'

W = mass * gravity * height
gravity acceleration = 9.754 m/s^2

V = W / Q

    >>> (Value('5 watt') / Value('2 coulomb')).as_string()
    '2.5 volt'

W = Q * V

    >>> (Value('3 coulomb') * Value('2.1 volt')).as_string()
    '6.3 watt'

Q = W / V

    >>> (Value('7 watt') / Value('3.5 volt')).as_string()
    '2 coulomb'

Useful life = Ampere/hour / Flow

    >>> (Value('7 ampere-hour') / Value('2 ampere')).as_string()
    '3.5 hour'

V = R . I

    >>> (Value('4 ohm') * Value('2.1 ampere')).as_string()
    '8.4 volt'

R = V / I

    >>> (Value('9 volt') / Value('2 ampere')).as_string()
    '4.5 ohm'

I = V / R

    >>> (Value('9 volt') / Value('2 ohm')).as_string()
    '4.5 ampere'

Exp x Exp

    >>> (Value('9x10^6') * Value('2x10^2')).as_string()
    '18x10^8'
